+    {
+        Throwable throwable = (Throwable) httpReq.getAttribute("javax.servlet.error.exception");
+        if (throwable != null)
+        {
+            String message = throwable.getMessage();
+            logger.error(message, throwable);

+        httpRes.setStatus(SC_INTERNAL_SERVER_ERROR);
